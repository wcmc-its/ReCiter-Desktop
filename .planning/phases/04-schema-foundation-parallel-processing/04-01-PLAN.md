    Per D-02 and RESEARCH Pattern 1. Replace the bare `FastAPI()` constructor with a lifespan-based startup:
    ```python
    from contextlib import asynccontextmanager
    from alembic.config import Config
    from alembic import command
    import os

    @asynccontextmanager
    async def lifespan(app):
        # Run Alembic migrations on startup (per D-02)
        cfg = Config()
        cfg.set_main_option(
            "script_location",
            os.path.join(os.path.dirname(__file__), "migrations")
        )
        cfg.set_main_option("sqlalchemy.url", os.environ.get(
            "DATABASE_URL",
            "mysql+pymysql://reciter:reciter_local@localhost:3306/reciter_desktop"
        ))
        command.upgrade(cfg, "head")
        yield

    app = FastAPI(title="ReCiter Desktop API", version="1.0.0", lifespan=lifespan)
    ```
